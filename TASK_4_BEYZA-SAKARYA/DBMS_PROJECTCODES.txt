        private void SearchInDataGrid(string searchText)
        {
            foreach (DataGridViewRow row in dataGridView1.Rows)
            {
                foreach (DataGridViewCell cell in row.Cells)
                {
                    if (cell.Value != null && cell.Value.ToString().Contains(searchText))
                    {
                        row.Selected = true;
                        return;
                    }
                }
            }
        }
